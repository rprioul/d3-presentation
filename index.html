Utilisation de D3


const svgSelection = d3.select(".circled3")
  .append("svg")
  .attr("width", 100)
  .attr("height", 100);

svgSelection.append("circle")
  .attr("cx", 50)
  .attr("cy", 50)
  .attr("r", 50)
  .style("fill", "purple");
